// --- SAFE IMAGE COMPONENT ---
const SafeImage = ({ src, alt, className, fallbackChar, theme }) => {
  const [error, setError] = useState(false);
  const [loaded, setLoaded] = useState(false);

  return (
    <div className={`${className} relative overflow-hidden bg-slate-800`}>
      {!loaded && !error && (
        <div className="absolute inset-0 bg-slate-700 animate-pulse" />
      )}
      {error ? (
        <div className="w-full h-full flex items-center justify-center font-black text-slate-600 text-lg select-none">
          {fallbackChar || '?'}
        </div>
      ) : (
        <img 
          src={src} 
          alt={alt} 
          className={`w-full h-full object-contain transition-opacity duration-300 ${loaded ? 'opacity-100' : 'opacity-0'}`} 
          onLoad={() => setLoaded(true)}
          onError={() => setError(true)} 
        />
      )}
    </div>
  );
};

// --- TOAST COMPONENT ---
const Toast = ({ msg, type, theme }) => (
  <div className="fixed top-20 left-1/2 -translate-x-1/2 z-[100] animate-in slide-in-from-top-4 fade-in pointer-events-none w-max max-w-[90%]">
    <div className={`px-6 py-3 rounded-full shadow-2xl flex items-center gap-3 border backdrop-blur-md ${
      type === 'error' 
        ? (theme === 'dark' ? 'bg-red-950/90 border-red-800 text-red-100' : 'bg-red-100 border-red-200 text-red-900') 
        : (theme === 'dark' ? 'bg-zinc-900/90 border-zinc-800 text-white' : 'bg-white/90 border-gray-200 text-black')
    }`}>
      {type === 'error' ? <X className="w-5 h-5"/> : <Check className="w-5 h-5 text-emerald-400"/>}
      <span className="font-bold text-sm truncate">{msg}</span>
    </div>
  </div>
);

// --- MAIN APPLICATION ---
export default function App() {
  const [user, setUser] = useState(null); 
  const [isAuthChecking, setIsAuthChecking] = useState(true);
  const [view, setView] = useState('join');
  const [lobbies, setLobbies] = useState([]);
  const [loadingLobbies, setLoadingLobbies] = useState(true);
  const [activeLobby, setActiveLobby] = useState(null);
  const [filter, setFilter] = useState('all');
  const [notification, setNotification] = useState(null);
  const [showPremium, setShowPremium] = useState(false);
  const [theme, setTheme] = useState('dark');
  const [showAutoJoin, setShowAutoJoin] = useState(false);

  // --- THEME ENGINE ---
  const t = {
    bg: theme === 'dark' ? 'bg-black' : 'bg-gray-50',
    text: theme === 'dark' ? 'text-white' : 'text-black',
    textSub: theme === 'dark' ? 'text-zinc-500' : 'text-gray-500',
    card: theme === 'dark' ? 'bg-zinc-900 border-zinc-800' : 'bg-white border-gray-200 shadow-sm',
    input: theme === 'dark' ? 'bg-black border-zinc-800 text-white' : 'bg-gray-50 border-gray-200 text-black',
    border: theme === 'dark' ? 'border-zinc-800' : 'border-gray-200',
    nav: theme === 'dark' ? 'bg-zinc-950 border-zinc-900' : 'bg-white border-gray-200',
    buttonPrimary: theme === 'dark' ? 'bg-white text-black hover:bg-zinc-200' : 'bg-black text-white hover:bg-gray-800',
    buttonSecondary: theme === 'dark' ? 'bg-zinc-900 text-white hover:bg-zinc-800' : 'bg-white text-black hover:bg-gray-100 border border-gray-200',
    modalOverlay: theme === 'dark' ? 'bg-black/90' : 'bg-white/90',
  };

  const notify = useCallback((msg, type = 'success') => {
    setNotification({ msg, type });
    setTimeout(() => setNotification(null), 3000);
  }, []);

  // --- PERSISTENCE ---
  useEffect(() => {
    try {
      const savedUser = localStorage.getItem('goRaidersUser_v2');
      if (savedUser) {
        setUser(JSON.parse(savedUser));
      }
    } catch (e) {
      console.error("Storage error", e);
    } finally {
      setIsAuthChecking(false);
    }
  }, []);

  useEffect(() => {
    try {
      if (user) {
        localStorage.setItem('goRaidersUser_v2', JSON.stringify(user));
      } else if (isAuthChecking === false) {
        // Only remove if we've finished checking and user is explicitly null (logged out)
        localStorage.removeItem('goRaidersUser_v2');
      }
    } catch (e) {
      console.error("Storage save error", e);
    }
  }, [user, isAuthChecking]);

  // --- DATA MOCK ---
  useEffect(() => {
    const generateLobbies = () => Array.from({ length: 8 }).map((_, i) => ({
      id: `lobby-${Date.now()}-${i}`,
      boss: BOSSES[i % BOSSES.length],
      host: `Trainer_${Math.floor(Math.random() * 9999)}`,
      hostRating: (3.5 + Math.random() * 1.5).toFixed(1),
      players: Array.from({ length: Math.floor(Math.random() * 4) + 1 }).map((_, j) => ({
        id: `p-${i}-${j}`, name: `Raider${j}`, level: 30 + j, team: 'valor', isReady: false
      })),
      maxPlayers: 6,
      team: TEAMS[i % 4].id,
      minLevel: i % 3 === 0 ? 40 : 1,
      weather: Math.random() > 0.6,
      createdAt: Date.now() - (Math.random() * 300000),
      timeLeft: Math.floor(Math.random() * 40) + 5
    }));

    setTimeout(() => {
      setLobbies(generateLobbies());
      setLoadingLobbies(false);
    }, 1000);

    const interval = setInterval(() => {
      setLobbies(prev => {
        const active = prev.filter(l => (Date.now() - l.createdAt) < 600000);
        if (Math.random() > 0.5) {
          const newBoss = BOSSES[Math.floor(Math.random() * BOSSES.length)];
          const newLobby = {
            id: `new-${Date.now()}`,
            boss: newBoss,
            host: `NewHost_${Math.floor(Math.random() * 100)}`,
            hostRating: '5.0',
            players: [{ id: 'host', name: 'NewHost', level: 50, team: 'mystic', isReady: true, isHost: true }],
            maxPlayers: 6,
            team: TEAMS[Math.floor(Math.random() * 4)].id,
            minLevel: 1,
            weather: false,
            createdAt: Date.now(),
            timeLeft: 45
          };
          return [newLobby, ...active].slice(0, 20);
        }
        return active;
      });
    }, 4000);
    return () => clearInterval(interval);
  }, []);

  // --- AUTO JOIN ---
  const executeAutoJoin = (boss) => {
    setShowAutoJoin(false);
    notify(`Searching for ${boss.name}...`);
    setTimeout(() => {
      const autoLobby = {
        id: `autojoin-${Date.now()}`,
        boss: boss,
        host: `Priority_Host`,
        hostRating: '5.0',
        players: [{ id: 'host', name: 'Priority_Host', level: 50, team: 'valor', isReady: true, isHost: true }],
        maxPlayers: 6,
        team: TEAMS[0].id, 
        minLevel: 1,
        weather: true,
        createdAt: Date.now(),
        timeLeft: 45
      };
      autoLobby.players.push({
        id: 'me',
        name: user.name,
        level: user.level,
        team: user.team,
        isReady: false
      });
      setActiveLobby(autoLobby);
      setView('lobby');
      notify(`Joined ${boss.name} Queue!`, "success");
    }, 1500);
  };

  if (isAuthChecking) return <div className="h-screen bg-black flex items-center justify-center text-white"><Loader2 className="w-8 h-8 animate-spin text-zinc-500"/></div>;

  return (
    <div className={`h-screen ${t.bg} ${t.text} font-sans flex flex-col overflow-hidden relative transition-colors duration-300`}>
      {!user ? (
        <Onboarding t={t} theme={theme} onComplete={(u) => { setUser(u); notify("Account Verified"); }} />
      ) : (
        <>
          <Header user={user} t={t} activeView={view} onPremium={() => setShowPremium(true)} />
          <main className="flex-1 overflow-y-auto scroll-smooth relative no-scrollbar">
            {view === 'join' && (
              <JoinFeed 
                lobbies={lobbies} 
                loading={loadingLobbies}
                filter={filter} 
                setFilter={setFilter} 
                user={user} 
                t={t}
                theme={theme}
                onJoin={(lobby) => { 
                  const myPlayer = { id: 'me', name: user.name, level: user.level, team: user.team, isReady: false };
                  const players = Array.isArray(lobby.players) ? lobby.players : [];
                  setActiveLobby({...lobby, isHost: false, players: [...players, myPlayer]}); 
                  setView('lobby'); 
                }}
                onNotify={notify}
                onPremium={() => setShowPremium(true)}
                onAutoJoin={() => user.isPremium ? setShowAutoJoin(true) : setShowPremium(true)}
              />
            )}
            {view === 'host' && (
              <HostView 
                user={user} 
                t={t}
                theme={theme}
                onHost={(newLobby) => { setLobbies([newLobby, ...lobbies]); setActiveLobby(newLobby); setView('lobby'); notify("Raid Published"); }} 
                onNotify={notify}
              />
            )}
            {view === 'shop' && <ShopView user={user} t={t} onUpgrade={() => setShowPremium(true)} />}
            {view === 'profile' && <ProfileView user={user} t={t} theme={theme} onLogout={() => { setUser(null); setView('join'); }} toggleTheme={() => setTheme(prev => prev === 'dark' ? 'light' : 'dark')} onPremium={() => setShowPremium(true)} />}
            {view === 'lobby' && activeLobby && (
              <LobbyView 
                lobby={activeLobby} 
                user={user} 
                t={t}
                theme={theme}
                updateLobby={setActiveLobby}
                onLeave={() => { setActiveLobby(null); setView('join'); }} 
                onNotify={notify} 
              />
            )}
          </main>
          <BottomNav currentView={view} setView={setView} t={t} />
        </>
      )}
      
      {showPremium && <PremiumModal t={t} onClose={() => setShowPremium(false)} onUpgrade={() => { setUser({...user, isPremium: true}); setShowPremium(false); notify("Welcome to Elite!"); }} />}
      {showAutoJoin && <AutoJoinModal t={t} theme={theme} onClose={() => setShowAutoJoin(false)} onSelect={executeAutoJoin} />}
      {notification && <Toast msg={notification.msg} type={notification.type} theme={theme} />}
    </div>
  );
}

// --- JOIN FEED ---
function JoinFeed({ lobbies, loading, filter, setFilter, user, onJoin, onNotify, onPremium, onAutoJoin, t, theme }) {
  // Skeleton Loader
  if (loading) return (
    <div className="p-4 space-y-4">
      <div className={`w-full h-24 ${t.card} animate-pulse rounded-3xl`}/>
      <div className="flex gap-2">{[1,2,3,4].map(i => <div key={i} className={`w-16 h-8 ${t.card} animate-pulse rounded-full`}/>)}</div>
      {[1,2,3].map(i => <div key={i} className={`w-full h-20 ${t.card} animate-pulse rounded-xl`}/>)}
    </div>
  );

  return (
    <div className="p-4 space-y-4 pb-24">
      {/* Auto Join */}
      <button onClick={onAutoJoin} className="w-full bg-gradient-to-r from-indigo-600 to-purple-600 p-4 rounded-3xl flex items-center justify-between shadow-xl relative overflow-hidden group active:scale-[0.98] transition-transform">
        <div className="flex items-center relative z-10">
          <div className="p-3 bg-white/20 rounded-2xl mr-4 backdrop-blur-sm"><Radar className="w-6 h-6 text-white" /></div>
          <div className="text-left text-white">
            <div className="font-black text-lg">Auto Join</div>
            <div className="text-xs opacity-80 font-medium">Find boss & skip queue instantly</div>
          </div>
        </div>
        <div className="bg-white text-indigo-600 text-xs font-black px-3 py-1 rounded-full relative z-10">{user.isPremium ? 'READY' : 'PRO'}</div>
      </button>

      {/* Filters */}
      <div className="flex gap-2 overflow-x-auto no-scrollbar py-1">
        {FILTERS.map(f => (
          <button key={f} onClick={() => setFilter(f)} className={`flex-shrink-0 px-4 py-2 rounded-full text-xs font-bold uppercase transition-all ${filter === f ? (theme === 'dark' ? 'bg-white text-black' : 'bg-black text-white') : `${t.card} ${t.textSub}`}`}>
            {f === 'max' ? 'MAX' : f}
          </button>
        ))}
      </div>

      {/* List */}
      <div className="space-y-2">
        {lobbies.map(lobby => {
          if (!lobby || !lobby.boss) return null;
          
          if (filter !== 'all') {
            if (filter === 'mega' && !lobby.boss.image.includes('mega')) return null;
            else if (filter === 'shadow' && !lobby.boss.isShadow) return null;
            else if (filter === 'max' && !lobby.boss.isDynamax) return null;
            else if (!['mega', 'shadow', 'max'].includes(filter) && lobby.boss.tier.toString() !== filter) return null;
          }

          const isLocked = !user.isPremium && (Date.now() - lobby.createdAt) < 10000;
          const team = TEAMS.find(t => t.id === lobby.team) || TEAMS[3];
          
          return (
            <div 
              key={lobby.id} 
              className={`${t.card} rounded-xl p-2 flex items-center space-x-3 relative overflow-hidden border-l-4 ${team.border} ${team.tint}`}
            >
              {isLocked && (
                <div className={`absolute inset-0 ${t.modalOverlay} z-20 flex flex-col items-center justify-center backdrop-blur-sm`}>
                  <Lock className={`w-5 h-5 ${t.textSub} mb-1`}/><span className={`text-[10px] font-bold ${t.textSub}`}>Wait 10s...</span>
                </div>
              )}
              
              <div className={`relative shrink-0 w-14 h-14 rounded-lg border ${t.border} flex items-center justify-center bg-black/5`}>

// --- LOBBY VIEW ---
function LobbyView({ lobby, user, onLeave, onNotify, t, theme }) {
  const [status, setStatus] = useState(lobby.status || 'waiting');
  const [players, setPlayers] = useState(Array.isArray(lobby.players) ? lobby.players : []);
  const [showRate, setShowRate] = useState(false);

  useEffect(() => {
    if (lobby.players && Array.isArray(lobby.players)) {
      setPlayers(lobby.players);
    }
  }, [lobby.players]);

  const handleKick = (id) => {
    const newPlayers = players.filter(p => p.id !== id);
    setPlayers(newPlayers);
    onNotify("Player Removed");
  };

  const handleLeave = () => {
    if (lobby.isHost) {
      onNotify("Lobby Closed");
      onLeave();
    } else {
      setShowRate(true);
    }
  };

  const handleReady = () => {
    setPlayers(prev => prev.map(p => p.id === 'me' ? {...p, isReady: !p.isReady} : p));
    setStatus('ready');
    onNotify("Host Notified!");
  };

  return (
    <div className="h-full flex flex-col p-6 relative">
      {/* RATING OVERLAY */}
      {showRate && (
        <div className={`absolute inset-0 z-50 ${t.modalOverlay} flex items-center justify-center p-6 animate-in fade-in backdrop-blur-sm`}>
          <div className={`${t.card} w-full max-w-sm rounded-[2rem] p-8 text-center border relative shadow-2xl`}>
            <h2 className="text-xl font-black mb-6">Rate Host</h2>
            <div className="flex justify-center gap-2 mb-8">
              {[1,2,3,4,5].map(s => (
                <button key={s} onClick={() => { onNotify("Rating Submitted"); onLeave(); }} className="p-2 hover:scale-110 transition-transform">
                  <Star className="w-8 h-8 text-amber-500 fill-current"/>
                </button>
              ))}
            </div>
            <button onClick={onLeave} className={`text-sm font-bold ${t.textSub}`}>Skip</button>
          </div>
        </div>
      )}

      <div className="flex-1 space-y-6 overflow-y-auto no-scrollbar">
        <div className="text-center relative">
          <div className="w-28 h-28 mx-auto relative mb-4">
            <div className={`absolute inset-0 rounded-full blur-3xl opacity-30 ${TEAMS.find(t => t.id === lobby.team)?.bg || 'bg-gray-500'}`}></div>
            <SafeImage src={lobby.boss.image} className="w-full h-full object-contain relative z-10" fallbackChar={lobby.boss.name[0]} theme={theme} />
          </div>
          <h2 className="text-3xl font-black uppercase tracking-tight">{lobby.boss.name}</h2>
          <div className="flex justify-center gap-2 mt-2">
            {lobby.weather && <span className="px-2 py-1 bg-sky-600 text-white text-[10px] font-bold rounded">WEATHER</span>}
            <span className={`px-2 py-1 ${t.input} text-[10px] font-bold rounded border ${t.border}`}>CP {lobby.boss.cp}</span>
          </div>
        </div>

        <div className={`${t.card} border rounded-3xl p-6 text-center space-y-6 shadow-xl`}>
          <div>
            <div className={`text-xs font-bold ${t.textSub} uppercase mb-2`}>Host Code</div>
            <div onClick={() => onNotify("Copied Code")} className={`text-3xl font-mono tracking-widest font-bold py-4 rounded-xl border ${t.border} ${t.input} cursor-pointer active:scale-95 transition-transform`}>
              {lobby.isHost ? user.code : '8821 0012 9918'}
            </div>
          </div>

          {lobby.isHost ? (
            <button onClick={() => { setStatus('sent'); onNotify("Players Notified!"); }} disabled={status === 'sent'} className={`w-full py-4 rounded-xl font-black flex items-center justify-center transition-all ${status === 'sent' ? 'bg-emerald-600 text-white' : t.buttonPrimary}`}>
              {status === 'sent' ? <Check className="mr-2 w-5 h-5"/> : <Send className="mr-2 w-5 h-5"/>}
              {status === 'sent' ? 'INVITES SENT' : 'NOTIFY PLAYERS'}
            </button>
          ) : (
            <button onClick={handleReady} disabled={status === 'ready'} className={`w-full py-4 rounded-xl font-black transition-all ${status === 'ready' ? 'bg-emerald-600/20 text-emerald-600 border border-emerald-600' : t.buttonPrimary}`}>
              {status === 'ready' ? 'WAITING FOR INVITE...' : 'I SENT REQUEST'}
            </button>
          )}
        </div>

        {/* PLAYER LIST */}
        <div>
          <div className={`text-xs font-bold ${t.textSub} uppercase mb-3 ml-2`}>Players ({players.length}/{lobby.maxPlayers})</div>
          <div className="space-y-2">
            {players.map((p) => {
              const pTeam = TEAMS.find(t => t.id === p.team) || TEAMS[3];
              const isMe = p.id === 'me';
              return (
                <div key={p.id} className={`${t.card} p-3 rounded-xl border flex items-center justify-between`}>
                  <div className="flex items-center gap-3">
                    <div className={`w-8 h-8 rounded-full border-2 ${pTeam.border} flex items-center justify-center font-bold text-xs ${t.bg}`}>
                      {p.name[0]}
                    </div>
                    <div>
                      <div className={`text-sm font-bold ${isMe ? 'text-indigo-500' : ''}`}>{p.name} {isMe && '(You)'}</div>
                      <div className={`text-[10px] ${t.textSub} font-medium`}>Level {p.level} • {pTeam.name}</div>
                    </div>
                  </div>
                  <div className="flex items-center gap-2">
                    {p.isReady && <CheckCircle2 className="w-5 h-5 text-emerald-500" />}
                    {lobby.isHost && !p.isHost && !isMe && (
                      <button onClick={() => handleKick(p.id)} className="p-2 text-red-500 hover:bg-red-500/10 rounded-lg transition-colors">
                        <UserMinus className="w-4 h-4" />
                      </button>
                    )}
                  </div>
                </div>
              );
            })}
            
            {/* Empty Slots */}
            {Array.from({ length: Math.max(0, lobby.maxPlayers - players.length) }).map((_, i) => (
              <div key={i} className={`p-3 rounded-xl border-2 border-dashed ${theme === 'dark' ? 'border-zinc-800' : 'border-gray-200'} flex items-center justify-center`}>
                <span className={`text-xs font-bold ${t.textSub} opacity-50`}>Waiting for player...</span>
              </div>
            ))}
          </div>
        </div>
      </div>

      <div className="space-y-3 mt-4 pt-4 border-t border-zinc-800/50">
        <button className={`w-full py-4 ${t.buttonSecondary} font-bold rounded-xl flex items-center justify-center transition-colors border ${t.border}`}>SWITCH TO GAME <ExternalLink className="ml-2 w-4 h-4"/></button>
        <button onClick={handleLeave} className="w-full py-3 text-red-500 font-bold text-xs uppercase hover:bg-red-500/10 rounded-lg">{lobby.isHost ? 'CLOSE LOBBY' : 'LEAVE RAID'}</button>
      </div>
    </div>
  );
}
